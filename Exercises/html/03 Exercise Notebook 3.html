# Your code here
def trigfit(x, y, freq):
    """Fit sum of sinusoids for y
    freq: Hz
    """
    X = np.column_stack(
                np.concatenate(
                    (
                        [np.sin(2 * np.pi * f * x) for f in freq],
                        [np.cos(2 * np.pi * f * x) for f in freq]
                    )
                )
    )

    # print(X[0])
    # print(X.shape)

    beta = np.linalg.inv(X.T.dot(X)).dot(X.T).dot(y)    # Least squares
    fit = X.dot(beta)                                   # Fit

    return beta, fit, y-fit
